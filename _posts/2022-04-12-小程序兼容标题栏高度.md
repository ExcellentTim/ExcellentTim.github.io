``` javascript
const res = wx.getSystemInfoSync()
if (res && res['system']) {
  let deviceHeight;
  if (res['system'].indexOf('Android') > 0) {
	deviceHeight = 48
  } else {
	deviceHeight = 44
  }
  const titleHeight = deviceHeight + res['statusBarHeight']
  this.setData({
	titleHeight: titleHeight + 'px'
  })
}
```
